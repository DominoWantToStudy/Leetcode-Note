#Is Subsequence
最佳代码
```cpp
class Solution {
public:
    bool isSubsequence(string s, string t) {
        if(s.length()==0)
            return true;
        if(t.length()==0)
            return false;
        int s_pos = 0;
        for(int i=0;i<t.length();++i){
            if(t[i]==s[s_pos]){
                ++s_pos;
            }
            if(s_pos==s.length())
                return true;
        }
        return false;
    }
};
```
